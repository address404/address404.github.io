```cpp
#include <iostream>
using namespace std;

class Circle
{
	int radius;

public:

	Circle(int r = 0){ this->radius = r; }
	int get_radius(){ return radius; }
	bool operator < (const Circle c){
		if (this->radius < c.radius) return true;
		else return false;
	}
};
```
